Backend Intelligence

Powered by Convex

Every Zivrodev project is backed by Convex — a reactive backend that syncs data in real-time, scales automatically, and eliminates the need for manual API management.

convex/schema.ts 
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  projects: defineTable({
    name: v.string(),
    status: v.string(),
    clientId: v.id("clients"),
    progress: v.number(),
  }),
  leads: defineTable({
    email: v.string(),
    source: v.string(),
    createdAt: v.number(),
  }),
  contacts: defineTable({
    name: v.string(),
    message: v.string(),
    phone: v.optional(v.string()),
  }),
});
main.js 
import { ConvexClient } from "convex/browser";
import { api } from "./convex/_generated/api";

const client = new ConvexClient(
  import.meta.env.VITE_CONVEX_URL
);

await client.mutation(api.leads.createLead, {
  email: "hello@zivrodev.com",
  source: "website"
});
